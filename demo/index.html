Quick Setup

Get running in under 10 lines.

main.ts 
import TailSlide, {
  ProgressPlugin,
  SlideNumberPlugin,
  OverviewPlugin
} from 'tailslide';

const deck = new TailSlide({
  transition: 'slide',
  dark: true,
});

deck.use(new ProgressPlugin());
deck.use(new SlideNumberPlugin());
deck.use(new OverviewPlugin());
